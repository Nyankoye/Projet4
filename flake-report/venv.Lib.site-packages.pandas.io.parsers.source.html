389 def validate_integer(name, val, min_val=0):
390 """
391 Checks whether the 'name' parameter for parsing is either
392 an integer OR float that can SAFELY be cast to an integer
393 without losing accuracy. Raises a ValueError if that is
394 not the case.
395  
396 Parameters
397 ----------
398 name : string
399 Parameter name (used for error reporting)
400 val : int or float
401 The value to check
402 min_val : int
403 Minimum allowed value (val < min_val will result in a ValueError)
404 """
405 msg = f"'{name:s}' must be an integer >={min_val:d}"
406  
407 if val is not None:
408 if is_float(val):
409 if int(val) != val:
410 raise ValueError(msg)
411 val = int(val)
412 elif not (is_integer(val) and val >= min_val):
413 raise ValueError(msg)
414  
415 return val
