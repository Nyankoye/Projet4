50 def assert_deprecated(self, function, num=1, ignore_others=False,
51 function_fails=False,
52 exceptions=np._NoValue,
53 args=(), kwargs={}):
54 """Test if DeprecationWarnings are given and raised.
55  
56 This first checks if the function when called gives `num`
57 DeprecationWarnings, after that it tries to raise these
58 DeprecationWarnings and compares them with `exceptions`.
59 The exceptions can be different for cases where this code path
60 is simply not anticipated and the exception is replaced.
61  
62 Parameters
63 ----------
64 function : callable
65 The function to test
66 num : int
67 Number of DeprecationWarnings to expect. This should normally be 1.
68 ignore_others : bool
69 Whether warnings of the wrong type should be ignored (note that
70 the message is not checked)
71 function_fails : bool
72 If the function would normally fail, setting this will check for
73 warnings inside a try/except block.
74 exceptions : Exception or tuple of Exceptions
75 Exception to expect when turning the warnings into an error.
76 The default checks for DeprecationWarnings. If exceptions is
77 empty the function is expected to run successfully.
78 args : tuple
79 Arguments for `function`
80 kwargs : dict
81 Keyword arguments for `function`
82 """
83 # reset the log
84 self.log[:] = []
85  
86 if exceptions is np._NoValue:
87 exceptions = (self.warning_cls,)
88  
89 try:
90 function(*args, **kwargs)
91 except (Exception if function_fails else tuple()):
92 pass
93  
94 # just in case, clear the registry
95 num_found = 0
96 for warning in self.log:
97 if warning.category is self.warning_cls:
98 num_found += 1
99 elif not ignore_others:
100 raise AssertionError(
101 "expected %s but got: %s" %
102 (self.warning_cls.__name__, warning.category))
103 if num is not None and num_found != num:
104 msg = "%i warnings found but %i expected." % (len(self.log), num)
105 lst = [str(w) for w in self.log]
106 raise AssertionError("\n".join([msg] + lst))
107  
108 with warnings.catch_warnings():
109 warnings.filterwarnings("error", message=self.message,
110 category=self.warning_cls)
111 try:
112 function(*args, **kwargs)
113 if exceptions != tuple():
114 raise AssertionError(
115 "No error raised during function call")
116 except exceptions:
117 if exceptions == tuple():
118 raise AssertionError(
119 "Error raised during function call")
