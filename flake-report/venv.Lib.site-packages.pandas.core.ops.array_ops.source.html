59 def _masked_arith_op(x: np.ndarray, y, op):
60 """
61 If the given arithmetic operation fails, attempt it again on
62 only the non-null elements of the input array(s).
63  
64 Parameters
65 ----------
66 x : np.ndarray
67 y : np.ndarray, Series, Index
68 op : binary operator
69 """
70 # For Series `x` is 1D so ravel() is a no-op; calling it anyway makes
71 # the logic valid for both Series and DataFrame ops.
72 xrav = x.ravel()
73 assert isinstance(x, np.ndarray), type(x)
74 if isinstance(y, np.ndarray):
75 dtype = find_common_type([x.dtype, y.dtype])
76 result = np.empty(x.size, dtype=dtype)
77  
78 if len(x) != len(y):
79 raise ValueError(x.shape, y.shape)
80 else:
81 ymask = notna(y)
82  
83 # NB: ravel() is only safe since y is ndarray; for e.g. PeriodIndex
84 # we would get int64 dtype, see GH#19956
85 yrav = y.ravel()
86 mask = notna(xrav) & ymask.ravel()
87  
88 # See GH#5284, GH#5035, GH#19448 for historical reference
89 if mask.any():
90 with np.errstate(all="ignore"):
91 result[mask] = op(xrav[mask], yrav[mask])
92  
93 else:
94 if not is_scalar(y):
95 raise TypeError(
96 f"Cannot broadcast np.ndarray with operand of type { type(y) }"
97 )
98  
99 # mask is only meaningful for x
100 result = np.empty(x.size, dtype=x.dtype)
101 mask = notna(xrav)
102  
103 # 1 ** np.nan is 1. So we have to unmask those.
104 if op is pow:
105 mask = np.where(x == 1, False, mask)
106 elif op is rpow:
107 mask = np.where(y == 1, False, mask)
108  
109 if mask.any():
110 with np.errstate(all="ignore"):
111 result[mask] = op(xrav[mask], y)
112  
113 result, _ = maybe_upcast_putmask(result, ~mask, np.nan)
114 result = result.reshape(x.shape) # 2D compat
115 return result

118 def _na_arithmetic_op(left, right, op, is_cmp: bool = False):
119 """
120 Return the result of evaluating op on the passed in values.
121  
122 If native types are not compatible, try coercion to object dtype.
123  
124 Parameters
125 ----------
126 left : np.ndarray
127 right : np.ndarray or scalar
128 is_cmp : bool, default False
129 If this a comparison operation.
130  
131 Returns
132 -------
133 array-like
134  
135 Raises
136 ------
137 TypeError : invalid operation
138 """
139 import pandas.core.computation.expressions as expressions
140  
141 try:
142 result = expressions.evaluate(op, left, right)
143 except TypeError:
144 if is_cmp:
145 # numexpr failed on comparison op, e.g. ndarray[float] > datetime
146 # In this case we do not fall back to the masked op, as that
147 # will handle complex numbers incorrectly, see GH#32047
148 raise
149 result = _masked_arith_op(left, right, op)
150  
151 if is_cmp and (is_scalar(result) or result is NotImplemented):
152 # numpy returned a scalar instead of operating element-wise
153 # e.g. numeric array vs str
154 return invalid_comparison(left, right, op)
155  
156 return missing.dispatch_fill_zeros(op, left, right, result)

159 def arithmetic_op(left: ArrayLike, right: Any, op):
160 """
161 Evaluate an arithmetic operation `+`, `-`, `*`, `/`, `//`, `%`, `**`, ...
162  
163 Parameters
164 ----------
165 left : np.ndarray or ExtensionArray
166 right : object
167 Cannot be a DataFrame or Index. Series is *not* excluded.
168 op : {operator.add, operator.sub, ...}
169 Or one of the reversed variants from roperator.
170  
171 Returns
172 -------
173 ndarray or ExtensionArray
174 Or a 2-tuple of these in the case of divmod or rdivmod.
175 """
176  
177 # NB: We assume that extract_array has already been called
178 # on `left` and `right`.
179 lvalues = ensure_wrapped_if_datetimelike(left)
180 rvalues = ensure_wrapped_if_datetimelike(right)
181 rvalues = _maybe_upcast_for_op(rvalues, lvalues.shape)
182  
183 if should_extension_dispatch(lvalues, rvalues) or isinstance(rvalues, Timedelta):
184 # Timedelta is included because numexpr will fail on it, see GH#31457
185 res_values = op(lvalues, rvalues)
186  
187 else:
188 with np.errstate(all="ignore"):
189 res_values = _na_arithmetic_op(lvalues, rvalues, op)
190  
191 return res_values

194 def comparison_op(left: ArrayLike, right: Any, op) -> ArrayLike:
195 """
196 Evaluate a comparison operation `=`, `!=`, `>=`, `>`, `<=`, or `<`.
197  
198 Parameters
199 ----------
200 left : np.ndarray or ExtensionArray
201 right : object
202 Cannot be a DataFrame, Series, or Index.
203 op : {operator.eq, operator.ne, operator.gt, operator.ge, operator.lt, operator.le}
204  
205 Returns
206 -------
207 ndarray or ExtensionArray
208 """
209 # NB: We assume extract_array has already been called on left and right
210 lvalues = ensure_wrapped_if_datetimelike(left)
211 rvalues = right
212  
213 rvalues = lib.item_from_zerodim(rvalues)
214 if isinstance(rvalues, list):
215 # TODO: same for tuples?
216 rvalues = np.asarray(rvalues)
217  
218 if isinstance(rvalues, (np.ndarray, ABCExtensionArray)):
219 # TODO: make this treatment consistent across ops and classes.
220 # We are not catching all listlikes here (e.g. frozenset, tuple)
221 # The ambiguous case is object-dtype. See GH#27803
222 if len(lvalues) != len(rvalues):
223 raise ValueError(
224 "Lengths must match to compare", lvalues.shape, rvalues.shape
225 )
226  
227 if should_extension_dispatch(lvalues, rvalues):
228 # Call the method on lvalues
229 res_values = op(lvalues, rvalues)
230  
231 elif is_scalar(rvalues) and isna(rvalues):
232 # numpy does not like comparisons vs None
233 if op is operator.ne:
234 res_values = np.ones(lvalues.shape, dtype=bool)
235 else:
236 res_values = np.zeros(lvalues.shape, dtype=bool)
237  
238 elif is_numeric_v_string_like(lvalues, rvalues):
239 # GH#36377 going through the numexpr path would incorrectly raise
240 return invalid_comparison(lvalues, rvalues, op)
241  
242 elif is_object_dtype(lvalues.dtype):
243 res_values = comp_method_OBJECT_ARRAY(op, lvalues, rvalues)
244  
245 else:
246 with warnings.catch_warnings():
247 # suppress warnings from numpy about element-wise comparison
248 warnings.simplefilter("ignore", DeprecationWarning)
249 with np.errstate(all="ignore"):
250 res_values = _na_arithmetic_op(lvalues, rvalues, op, is_cmp=True)
251  
252 return res_values

255 def na_logical_op(x: np.ndarray, y, op):
256 try:
257 # For exposition, write:
258 # yarr = isinstance(y, np.ndarray)
259 # yint = is_integer(y) or (yarr and y.dtype.kind == "i")
260 # ybool = is_bool(y) or (yarr and y.dtype.kind == "b")
261 # xint = x.dtype.kind == "i"
262 # xbool = x.dtype.kind == "b"
263 # Then Cases where this goes through without raising include:
264 # (xint or xbool) and (yint or bool)
265 result = op(x, y)
266 except TypeError:
267 if isinstance(y, np.ndarray):
268 # bool-bool dtype operations should be OK, should not get here
269 assert not (is_bool_dtype(x.dtype) and is_bool_dtype(y.dtype))
270 x = ensure_object(x)
271 y = ensure_object(y)
272 result = libops.vec_binop(x.ravel(), y.ravel(), op)
273 else:
274 # let null fall thru
275 assert lib.is_scalar(y)
276 if not isna(y):
277 y = bool(y)
278 try:
279 result = libops.scalar_binop(x, y, op)
280 except (
281 TypeError,
282 ValueError,
283 AttributeError,
284 OverflowError,
285 NotImplementedError,
286 ) as err:
287 typ = type(y).__name__
288 raise TypeError(
289 f"Cannot perform '{op.__name__}' with a dtyped [{x.dtype}] array "
290 f"and scalar of type [{typ}]"
291 ) from err
292  
293 return result.reshape(x.shape)

296 def logical_op(left: ArrayLike, right: Any, op) -> ArrayLike:
297 """
298 Evaluate a logical operation `|`, `&`, or `^`.
299  
300 Parameters
301 ----------
302 left : np.ndarray or ExtensionArray
303 right : object
304 Cannot be a DataFrame, Series, or Index.
305 op : {operator.and_, operator.or_, operator.xor}
306 Or one of the reversed variants from roperator.
307  
308 Returns
309 -------
310 ndarray or ExtensionArray
311 """
  • E731 Do not assign a lambda expression, use a def

314 def fill_bool(x, left=None):
315 # if `left` is specifically not-boolean, we do not cast to bool
316 if x.dtype.kind in ["c", "f", "O"]:
317 # dtypes that can hold NA
318 mask = isna(x)
319 if mask.any():
320 x = x.astype(object)
321 x[mask] = False
322  
323 if left is None or is_bool_dtype(left.dtype):
324 x = x.astype(bool)
325 return x
326  
327 is_self_int_dtype = is_integer_dtype(left.dtype)
328  
329 right = lib.item_from_zerodim(right)
330 if is_list_like(right) and not hasattr(right, "dtype"):
331 # e.g. list, tuple
332 right = construct_1d_object_array_from_listlike(right)
333  
334 # NB: We assume extract_array has already been called on left and right
335 lvalues = ensure_wrapped_if_datetimelike(left)
336 rvalues = right
337  
338 if should_extension_dispatch(lvalues, rvalues):
339 # Call the method on lvalues
340 res_values = op(lvalues, rvalues)
341  
342 else:
343 if isinstance(rvalues, np.ndarray):
344 is_other_int_dtype = is_integer_dtype(rvalues.dtype)
345 rvalues = rvalues if is_other_int_dtype else fill_bool(rvalues, lvalues)
346  
347 else:
348 # i.e. scalar
349 is_other_int_dtype = lib.is_integer(rvalues)
350  
351 # For int vs int `^`, `|`, `&` are bitwise operators and return
352 # integer dtypes. Otherwise these are boolean ops
353 filler = fill_int if is_self_int_dtype and is_other_int_dtype else fill_bool
354  
355 res_values = na_logical_op(lvalues, rvalues, op)
356 # error: Cannot call function of unknown type
357 res_values = filler(res_values) # type: ignore[operator]
358  
359 return res_values

362 def get_array_op(op):
363 """
364 Return a binary array operation corresponding to the given operator op.
365  
366 Parameters
367 ----------
368 op : function
369 Binary operator from operator or roperator module.
370  
371 Returns
372 -------
373 functools.partial
374 """
375 if isinstance(op, partial):
376 # We get here via dispatch_to_series in DataFrame case
377 # TODO: avoid getting here
378 return op
379  
380 op_name = op.__name__.strip("_").lstrip("r")
381 if op_name == "arith_op":
382 # Reached via DataFrame._combine_frame
383 return op
384  
385 if op_name in {"eq", "ne", "lt", "le", "gt", "ge"}:
386 return partial(comparison_op, op=op)
387 elif op_name in {"and", "or", "xor", "rand", "ror", "rxor"}:
388 return partial(logical_op, op=op)
389 elif op_name in {
390 "add",
391 "sub",
392 "mul",
393 "truediv",
394 "floordiv",
395 "mod",
396 "divmod",
397 "pow",
398 }:
399 return partial(arithmetic_op, op=op)
400 else:
401 raise NotImplementedError(op_name)

404 def _maybe_upcast_for_op(obj, shape: Shape):
405 """
406 Cast non-pandas objects to pandas types to unify behavior of arithmetic
407 and comparison operations.
408  
409 Parameters
410 ----------
411 obj: object
412 shape : tuple[int]
413  
414 Returns
415 -------
416 out : object
417  
418 Notes
419 -----
420 Be careful to call this *after* determining the `name` attribute to be
421 attached to the result of the arithmetic operation.
422 """
423 from pandas.core.arrays import DatetimeArray, TimedeltaArray
424  
425 if type(obj) is timedelta:
426 # GH#22390 cast up to Timedelta to rely on Timedelta
427 # implementation; otherwise operation against numeric-dtype
428 # raises TypeError
429 return Timedelta(obj)
430 elif isinstance(obj, np.datetime64):
431 # GH#28080 numpy casts integer-dtype to datetime64 when doing
432 # array[int] + datetime64, which we do not allow
433 if isna(obj):
434 # Avoid possible ambiguities with pd.NaT
435 obj = obj.astype("datetime64[ns]")
436 right = np.broadcast_to(obj, shape)
437 return DatetimeArray(right)
438  
439 return Timestamp(obj)
440  
441 elif isinstance(obj, np.timedelta64):
442 if isna(obj):
443 # wrapping timedelta64("NaT") in Timedelta returns NaT,
444 # which would incorrectly be treated as a datetime-NaT, so
445 # we broadcast and wrap in a TimedeltaArray
446 obj = obj.astype("timedelta64[ns]")
447 right = np.broadcast_to(obj, shape)
448 return TimedeltaArray(right)
449  
450 # In particular non-nanosecond timedelta64 needs to be cast to
451 # nanoseconds, or else we get undesired behavior like
452 # np.timedelta64(3, 'D') / 2 == np.timedelta64(1, 'D')
453 return Timedelta(obj)
454  
455 elif isinstance(obj, np.ndarray) and obj.dtype.kind == "m":
456 # GH#22390 Unfortunately we need to special-case right-hand
457 # timedelta64 dtypes because numpy casts integer dtypes to
458 # timedelta64 when operating with timedelta64
459 return TimedeltaArray._from_sequence(obj)
460 return obj
